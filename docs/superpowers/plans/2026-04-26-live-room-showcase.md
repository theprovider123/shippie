# Live Room — Showcase App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A four-to-eight-player real-time pub-quiz app that demonstrates the integration of every Shippie platform layer in one experience: mesh networking (proximity), sensory textures (the buzzer feel + scoreboard pop + winner celebration), the observer runtime (auto-applied haptics on every tap), and the wrapper PWA shell (installable on a phone, works offline within the local mesh).

**Architecture:** One device hosts a room and broadcasts a six-character join code. Other devices join via the code (QR or typed). The host advances through three sample questions; guests buzz in to answer. First-buzzer-wins lockout is enforced via CRDT (Yjs `Y.Map`) — concurrent buzzes resolve deterministically by Lamport timestamp tiebreak (lowest peer-id wins on equal stamp). Scores update in a Yjs `Y.Array<{peerId, points}>`; the scoreboard component subscribes and animates score deltas with spring physics. At end-of-quiz, the winning peer's row fires the `milestone` texture from Plan C.

**Tech Stack:** Vite + React 19 + TypeScript (matches showcase-whiteboard exactly), `@shippie/proximity` (mesh, already shipped 90/90 tests), `@shippie/sdk` (wrapper + textures from Plan C), Yjs (CRDT — already used by whiteboard), bun:test for unit tests, manual two-tab smoke for integration.

The CRDT schema:
- `Y.Map`: `currentIndex` (number), `phase` (`'lobby' | 'question' | 'reveal' | 'finished'`)
- `Y.Array<Buzz>`: append-only — `{ peerId, ts, questionIndex }`
- `Y.Array<Score>`: shared — `{ peerId, points }`

First-wins resolution: when reading buzzes for the current question, sort by `ts` ascending then by `peerId` ascending; the first entry is the winner. Lockout is a UI concept — once a buzz exists, every guest's button disables.

// apps/showcase-live-room/src/shared/quiz-state.ts
/**
 * Quiz state as a Yjs document. Three shared structures:
 *   - 'meta' Y.Map         currentIndex, phase
 *   - 'buzzes' Y.Array     append-only Buzz log (one entry per peer per question)
 *   - 'scores' Y.Array     shared Score table (mutated by host on award)
 *
 * First-buzzer-wins is computed from the buzz log on read, never persisted as
 * a separate "winner" field — keeping the source of truth in one place
 * removes a class of inconsistency bugs across peers.
 */
import * as Y from 'yjs';
import type { Buzz, Score, Phase } from './types.ts';

export function createQuizState(): Y.Doc {
  const doc = new Y.Doc();
  const meta = doc.getMap('meta');
  if (!meta.has('currentIndex')) meta.set('currentIndex', 0);
  if (!meta.has('phase')) meta.set('phase', 'lobby' satisfies Phase);
  return doc;
}

export function getMeta(doc: Y.Doc) {
  return doc.getMap('meta') as Y.Map<unknown>;
}

export function getCurrentIndex(doc: Y.Doc): number {
  return (getMeta(doc).get('currentIndex') as number) ?? 0;
}

export function setCurrentIndex(doc: Y.Doc, index: number): void {
  getMeta(doc).set('currentIndex', index);
}

export function getPhase(doc: Y.Doc): Phase {
  return (getMeta(doc).get('phase') as Phase) ?? 'lobby';
}

export function setPhase(doc: Y.Doc, phase: Phase): void {
  getMeta(doc).set('phase', phase);
}

export function recordBuzz(doc: Y.Doc, buzz: Buzz): void {
  const buzzes = doc.getArray<Buzz>('buzzes');
  buzzes.push([buzz]);
}

export function buzzesForQuestion(doc: Y.Doc, questionIndex: number): Buzz[] {
  const buzzes = doc.getArray<Buzz>('buzzes').toArray();
  return buzzes.filter((b) => b.questionIndex === questionIndex);
}

export function firstBuzzerForQuestion(doc: Y.Doc, questionIndex: number): Buzz | null {
  const candidates = buzzesForQuestion(doc, questionIndex);
  if (candidates.length === 0) return null;
  candidates.sort((a, b) => a.ts - b.ts || a.peerId.localeCompare(b.peerId));
  return candidates[0]!;
}

export function awardPointToWinner(doc: Y.Doc, questionIndex: number): void {
  const winner = firstBuzzerForQuestion(doc, questionIndex);
  if (!winner) return;
  const scores = doc.getArray<Score>('scores');
  const list = scores.toArray();
  const existing = list.findIndex((s) => s.peerId === winner.peerId);
  if (existing >= 0) {
    const current = list[existing]!;
    scores.delete(existing, 1);
    scores.insert(existing, [{ peerId: current.peerId, points: current.points + 1 }]);
  } else {
    scores.push([{ peerId: winner.peerId, points: 1 }]);
  }
}
```

import * as Y from 'yjs';
import {
  setCurrentIndex,
  setPhase,
  awardPointToWinner,
  getCurrentIndex,
  getPhase,
} from '../shared/quiz-state.ts';
import { QUESTIONS } from '../shared/questions.ts';

/** Host-only: orchestrate quiz progression. Guests are read-only on phase + currentIndex. */
export function startQuiz(doc: Y.Doc): void {
  setCurrentIndex(doc, 0);
  setPhase(doc, 'question');
}

export function reveal(doc: Y.Doc): void {
  if (getPhase(doc) !== 'question') return;
  awardPointToWinner(doc, getCurrentIndex(doc));
  setPhase(doc, 'reveal');
}

export function next(doc: Y.Doc): void {
  const index = getCurrentIndex(doc);
  if (index >= QUESTIONS.length - 1) {
    setPhase(doc, 'finished');
    return;
  }
  setCurrentIndex(doc, index + 1);
  setPhase(doc, 'question');
}

// apps/showcase-live-room/src/host/HostRoom.tsx
import { useEffect, useState, useSyncExternalStore } from 'react';
import * as Y from 'yjs';
import { shippie } from '@shippie/sdk';
import { ShareCode } from './ShareCode.tsx';
import { startQuiz, reveal, next } from './quiz-controller.ts';
import { getCurrentIndex, getPhase, firstBuzzerForQuestion } from '../shared/quiz-state.ts';
import { QUESTIONS } from '../shared/questions.ts';
import { Scoreboard } from '../shared/Scoreboard.tsx';

export function HostRoom() {
  const [room, setRoom] = useState<{ joinCode: string; doc: Y.Doc } | null>(null);

  useEffect(() => {
    let cancelled = false;
    void shippie.local.group.create({ appSlug: 'live-room' }).then((r) => {
      if (cancelled) return;
      setRoom({ joinCode: r.joinCode, doc: r.doc });
    });
    return () => { cancelled = true; };
  }, []);

  if (!room) return <p>Creating room…</p>;

  return <HostRoomReady room={room} />;
}

function HostRoomReady({ room }: { room: { joinCode: string; doc: Y.Doc } }) {
  const phase = useYjsField(room.doc, getPhase);
  const index = useYjsField(room.doc, getCurrentIndex);
  const winner = useYjsField(room.doc, (d) => firstBuzzerForQuestion(d, index));
  const question = QUESTIONS[index];

  return (
    <main className="host-room">
      {phase === 'lobby' && (
        <>
          <ShareCode code={room.joinCode} />
          <button type="button" onClick={() => startQuiz(room.doc)}>Start quiz</button>
        </>
      )}
      {phase === 'question' && question && (
        <>
          <h2>Q{index + 1}: {question.prompt}</h2>
          {winner && <p>Buzzed: <strong>{winner.peerId.slice(0, 6)}</strong></p>}
          <button type="button" onClick={() => reveal(room.doc)} disabled={!winner}>Reveal answer</button>
        </>
      )}
      {phase === 'reveal' && question && (
        <>
          <h2>{question.answer}</h2>
          <button type="button" onClick={() => next(room.doc)}>Next</button>
        </>
      )}
      {phase === 'finished' && <Scoreboard doc={room.doc} highlightWinner />}
    </main>
  );
}

/** Subscribe a React component to a derived Y.Doc value. */
function useYjsField<T>(doc: Y.Doc, read: (doc: Y.Doc) => T): T {
  return useSyncExternalStore(
    (cb) => {
      const handler = () => cb();
      doc.on('update', handler);
      return () => doc.off('update', handler);
    },
    () => read(doc),
    () => read(doc),
  );
}
```

import { useEffect, useState, useSyncExternalStore } from 'react';
import * as Y from 'yjs';
import { shippie } from '@shippie/sdk';
import { generatePeerId } from '../shared/peer-id.ts';
import { Buzzer } from './Buzzer.tsx';
import { Scoreboard } from '../shared/Scoreboard.tsx';
import { getCurrentIndex, getPhase, firstBuzzerForQuestion, recordBuzz } from '../shared/quiz-state.ts';
import { QUESTIONS } from '../shared/questions.ts';

interface GuestRoomProps { joinCode: string }

const peerId = generatePeerId();

export function GuestRoom({ joinCode }: GuestRoomProps) {
  const [doc, setDoc] = useState<Y.Doc | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    void shippie.local.group.join({ appSlug: 'live-room', joinCode })
      .then((r) => { if (!cancelled) setDoc(r.doc); })
      .catch((e: unknown) => { if (!cancelled) setError(e instanceof Error ? e.message : String(e)); });
    return () => { cancelled = true; };
  }, [joinCode]);

  if (error) return <p>Could not join: {error}</p>;
  if (!doc) return <p>Joining…</p>;

  return <GuestRoomReady doc={doc} />;
}

function GuestRoomReady({ doc }: { doc: Y.Doc }) {
  const phase = useYjsField(doc, getPhase);
  const index = useYjsField(doc, getCurrentIndex);
  const winner = useYjsField(doc, (d) => firstBuzzerForQuestion(d, index));
  const question = QUESTIONS[index];

  if (phase === 'lobby') return <p>Waiting for the host to start…</p>;
  if (phase === 'finished') return <Scoreboard doc={doc} highlightWinner />;
  if (phase === 'reveal' && question) return <p>Answer: <strong>{question.answer}</strong></p>;
  if (phase === 'question' && question) {
    const myWin = winner?.peerId === peerId;
    const someoneElseBuzzed = winner !== null && winner.peerId !== peerId;
    return (
      <div>
        <h2>Q{index + 1}: {question.prompt}</h2>
        <Buzzer
          locked={someoneElseBuzzed || myWin}
          onBuzz={() => recordBuzz(doc, { peerId, ts: Date.now(), questionIndex: index })}
          isMine={myWin}
        />
      </div>
    );
  }
  return null;
}

function useYjsField<T>(doc: Y.Doc, read: (doc: Y.Doc) => T): T {
  return useSyncExternalStore(
    (cb) => { const h = () => cb(); doc.on('update', h); return () => doc.off('update', h); },
    () => read(doc),
    () => read(doc),
  );
}

## Task 6: Scoreboard with spring physics

**Files:**
- Create: `apps/showcase-live-room/src/shared/Scoreboard.tsx`

Spring-animated horizontal bars. Each row's bar width interpolates with `animateSpring` from `@shippie/sdk/wrapper/spring`. The row with the highest score fires `milestone` once when the quiz is finished.

- [ ] **Step 1: Write `Scoreboard.tsx`**

```tsx
// apps/showcase-live-room/src/shared/Scoreboard.tsx
import { useEffect, useRef, useSyncExternalStore } from 'react';
import * as Y from 'yjs';
import { animateSpring, fireTexture } from '@shippie/sdk/wrapper';
import type { Score } from './types.ts';

interface ScoreboardProps {
  doc: Y.Doc;
  highlightWinner?: boolean;
}

export function Scoreboard({ doc, highlightWinner }: ScoreboardProps) {
  const scores = useScores(doc);
  const sorted = [...scores].sort((a, b) => b.points - a.points);
  const max = sorted[0]?.points ?? 1;
  const winnerId = highlightWinner ? sorted[0]?.peerId : null;
  const milestoneFired = useRef(false);

  useEffect(() => {
    if (winnerId && !milestoneFired.current) {
      milestoneFired.current = true;
      fireTexture('milestone', document.body);
    }
  }, [winnerId]);

  return (
    <ol className="scoreboard">
      {sorted.map((s) => (
        <ScoreRow key={s.peerId} score={s} max={max} isWinner={s.peerId === winnerId} />
      ))}
    </ol>
  );
}

function ScoreRow({ score, max, isWinner }: { score: Score; max: number; isWinner: boolean }) {
  const barRef = useRef<HTMLDivElement | null>(null);
  const target = score.points / Math.max(1, max);

  useEffect(() => {
    if (!barRef.current) return;
    animateSpring(({ value }) => {
      if (barRef.current) barRef.current.style.width = `${value * 100}%`;
    }, { from: 0, to: target, stiffness: 180, damping: 22 });
  }, [target]);

  return (
    <li className={`score-row ${isWinner ? 'score-row-winner' : ''}`}>
      <span className="score-peer">{score.peerId.slice(0, 6)}</span>
      <div className="score-bar-track">
        <div ref={barRef} className="score-bar" />
      </div>
      <span className="score-points">{score.points}</span>
    </li>
  );
}

function useScores(doc: Y.Doc): Score[] {
  return useSyncExternalStore(
    (cb) => { const h = () => cb(); doc.on('update', h); return () => doc.off('update', h); },
    () => doc.getArray<Score>('scores').toArray(),
    () => doc.getArray<Score>('scores').toArray(),
  );
}
```

- [ ] **Step 1: Write the test**

```typescript
// apps/showcase-live-room/src/integration.test.ts
import { describe, expect, it } from 'bun:test';
import * as Y from 'yjs';
import {
  createQuizState,
  recordBuzz,
  setCurrentIndex,
  setPhase,
  awardPointToWinner,
  firstBuzzerForQuestion,
} from './shared/quiz-state.ts';
import { startQuiz, reveal, next } from './host/quiz-controller.ts';

function sync(...docs: Y.Doc[]) {
  for (const a of docs) {
    for (const b of docs) {
      if (a === b) continue;
      Y.applyUpdate(b, Y.encodeStateAsUpdate(a));
    }
  }
}

describe('two-peer convergence', () => {
  it('host advances quiz; guest sees same phase + question', () => {
    const host = createQuizState();
    const guest = createQuizState();
    startQuiz(host);
    sync(host, guest);
    expect(guest.getMap('meta').get('phase')).toBe('question');
    expect(guest.getMap('meta').get('currentIndex')).toBe(0);
  });

  it('two guests buzz; deterministic winner across all peers', () => {
    const host = createQuizState();
    const g1 = createQuizState();
    const g2 = createQuizState();
    startQuiz(host);
    sync(host, g1, g2);

    recordBuzz(g1, { peerId: 'g1', ts: 100, questionIndex: 0 });
    recordBuzz(g2, { peerId: 'g2', ts: 100, questionIndex: 0 });
    sync(host, g1, g2);

    expect(firstBuzzerForQuestion(host, 0)?.peerId).toBe('g1'); // tiebreak
    expect(firstBuzzerForQuestion(g1, 0)?.peerId).toBe('g1');
    expect(firstBuzzerForQuestion(g2, 0)?.peerId).toBe('g1');
  });

  it('host awards point; all peers see updated score', () => {
    const host = createQuizState();
    const guest = createQuizState();
    startQuiz(host);
    recordBuzz(host, { peerId: 'g1', ts: 100, questionIndex: 0 });
    sync(host, guest);
    awardPointToWinner(host, 0);
    sync(host, guest);
    const guestScores = guest.getArray('scores').toArray() as Array<{ peerId: string; points: number }>;
    expect(guestScores).toEqual([{ peerId: 'g1', points: 1 }]);
  });

  it('full game flow ends in finished phase', () => {
    const host = createQuizState();
    startQuiz(host);
    for (let i = 0; i < 3; i++) {
      recordBuzz(host, { peerId: 'g1', ts: 100 + i, questionIndex: i });
      reveal(host);
      next(host);
    }
    expect(host.getMap('meta').get('phase')).toBe('finished');
    const scores = host.getArray('scores').toArray() as Array<{ peerId: string; points: number }>;
    expect(scores.find((s) => s.peerId === 'g1')?.points).toBe(3);
  });
});
```
